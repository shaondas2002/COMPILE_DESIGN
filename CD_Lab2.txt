#include <iostream>
using namespace std;

int main() {
    string input;
    cout << "Enter a value: ";
    cin >> input;

    int i;
    for (i = 0; i < input.length(); i++) {
        if (input[i] < 48 || input[i] > 57) {
            break;
        }
    }

    if (i == input.length()) {
        cout << "Numeric constant" << endl;
    } else {
        cout << "Not numeric" << endl;
    }

    return 0;
}

#include <iostream>
using namespace std;

int main() {
    string input;
    cout << "Enter an expression: ";
    cin >> input;


    char operators[] = {'+', '-', '*', '/', '%', '='};

    int operatorCount = 1;

    for (int i = 0; i < input.length(); i++) {
        for (int j = 0; j < 6; j++) {
            if (input[i] == operators[j]) {
                cout << "operator" << operatorCount << ": " << input[i] << endl;
                operatorCount++;
            }
        }
    }

    if (operatorCount == 1) {
        cout << "No arithmetic operators found." << endl;
    }

    return 0;
}

#include <iostream>
#include <string>
using namespace std;

int main() {
    string line;

    cout << "Enter lines of code (type 'END' to stop):" << endl;

    while (true) {
        getline(cin, line);

        if (line == "END") {
            break;
        }


        if (line.find("//") == 0) {
            cout << "single line comment." << endl;
        }

        else if (line.find("/*") == 0 && line.find("*/") != string::npos) {
            cout << "Multiple line comment." << endl;
        }
        else {
            cout << "This is NOT a comment line." << endl;
        }
    }

    return 0;
}

#include <iostream>
#include <string>
#include <cctype>

using namespace std;

int isValidIdentifier(const string& str) {
    int flag = 1;

    if (str.empty()) {
        flag = 0;
    }
    else if (!(isalpha(str[0]) || str[0] == '_')) {
        flag = 0;
    } else {
        for (size_t i = 1; i < str.length(); ++i) {
            if (!(isalnum(str[i]) || str[i] == '_')) {
                flag = 0;
                break;
            }
        }
    }

    return flag;
}

int main() {
    string input;

    cout << "Enter a string to check if it's a valid identifier: ";
    cin >> input;

    if (isValidIdentifier(input) == 1) {
        cout << "\"" << input << "\" is a valid identifier." << endl;
    } else {
        cout << "\"" << input << "\" is NOT a valid identifier." << endl;
    }

    return 0;
}
